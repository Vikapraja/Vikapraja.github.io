            //error.cause   :
            
            try {
                var a = 10
                var b = 0
                if (b === 0)
                    throw new Error("We Can't Divide a Number with 0", {
                        cause: { result: "Fail", resone: "Can't Divide By Zero Exception" },
                    });
                else
                    console.log(a / b)
            } catch (error) {
                console.log(error.cause)
            }
