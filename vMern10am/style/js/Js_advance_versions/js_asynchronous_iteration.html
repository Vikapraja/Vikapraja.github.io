[+] Js Asynchronous Iteration
[ECMA Script 2018]

yield = return (keyword) => Function will be run again from previous yield 

Example :
            async function* test(){
                setTimeout(()=>{
                    console.log("Hello World1")
                },1000)
                yield 1
                
                setTimeout(()=>{
                    console.log("Hello World2")
                },2000)
                yield 2
                
                setTimeout(()=>{
                    console.log("Hello World3")
                },3000)
                yield 3
                
                setTimeout(()=>{
                    console.log("Hello World4")
                },4000)
                yield 4
                
                setTimeout(()=>{
                    console.log("Hello World5")
                },5000)
                yield 5
            }
                
            async function sample(){
                for await(let i of test()){
                    console.log(i)
                }
            }
            sample()
