[+] Js Optional Chaining (?.) in object
[ECMA Script 2020]

Example :
            Optional Chaining(?.) :
                    var emp = {
                        id:1001,
                        name:"Nitin Chauhan",
                        dsg:"Trainer"
                    }
                    console.log(emp.id??"N/A")
                    console.log(emp.name??"N/A")
                    console.log(emp.dsg??"N/A")
                    console.log(emp.salary??"N/A")
                    console.log(emp.address?.city??"N/A")
