#If else example :

        if else :
                    var num: number = Number(prompt("Enter the Number : "))
                    var n: number = Math.floor(Math.sqrt(num))
                    if (n * n == num)
                            document.write(`${num} is Perfect Square Number`)
                    else
                            document.write(`${num} is not a Perfect Square Number`)



    Nested if else :
                    var num: number = Number(prompt("Enter the Number : "))
                    if (num % 2 == 0) {
                            if (num % 3 == 0)
                                    document.write(`${num} is Divisible by Both 2 and 3`)
                            else
                                    document.write(`${num} is Divisible by 2 but not Divisible by 3`)
                    }
                    else {
                            if (num % 3 == 0)
                                    document.write(`${num} is Divisible by 3 but not Divisible by 2`)
                            else
                                    document.write(`${num} is not Divisible by Both 2 and 3`)
                    }



    if else ladder
                    var a: number = Number(prompt("Enter the First Number : "))
                    var b: number = Number(prompt("Enter the Second Number : "))
                    var c: number = Number(prompt("Enter the Third Number : "))
                    var d: number = Number(prompt("Enter the Fourth Number : "))
                    var e: number = Number(prompt("Enter the Fifth Number : "))
                    if (a >= b && a >= c && a >= d && a >= e)
                            document.write(`${a} is Greatest`)
                    else if (b >= c && b >= d && b >= e)
                            document.write(`${b} is Greatest`)
                    else if (c >= d && c >= e)
                            document.write(`${c} is Greatest`)
                    else if (d >= e)
                            document.write(`${d} is Greatest`)
                    else
                            document.write(`${e} is Greatest`)


            


#Loops

        for loop :
                    var num: number = Number(prompt("Enter the Number : "))
                    var flag: boolean = false
                    for (let i = 2; i <= num ** 0.5; i++) {
                            if (num % i == 0) {
                                    flag = true
                                    break
                            }
                    }
                    if (flag == false && num >= 2)
                            document.write(`${num} is Prime Number`)
                    else
                            document.write(`${num} is not a Prime Number`)

        

        while loop :
                    var num:number = Number(prompt("Enter the Number : "))
                    var sum:number = 0
                    while(num!=0){
                            let r:number = num%10
                            sum = sum + r
                            num = Math.floor(num/10)
                    }
                    document.write(`sum = ${sum}`)

        

        do-while loop :
                    let i:number = 1
                    do{
                            document.write(`Hello World ${i}`)
                            i++
                    }while(i<=10)

        




Switch case: 

                var a: number = Number(prompt("Enter the First Number : "))
                var b: number = Number(prompt("Enter the Second Number : "))
                var ch: string = prompt("Press + for Addition\nPress - for Subtration\nPress * for Multiplication\nPress / for Division\nPress % for Remainder\nPress ** for Power\nEnter Your Choice  : ")
                switch (ch) {
                        case "+":
                                document.write(`${a} + ${b} = ${a + b}`)
                                break
                        case "-":
                                document.write(`${a} - ${b} = ${a - b}`)
                                break
                        case "*":
                                document.write(`${a} * ${b} = ${a * b}`)
                                break
                        case "/":
                                document.write(`${a} / ${b} = ${a / b}`)
                                break
                        case "%":
                                document.write(`${a} % ${b} = ${a % b}`)
                                break
                        case "**":
                                document.write(`${a} ** ${b} = ${a ** b}`)
                                break
                        default:
                                document.write(`Invalid Choice`)
                                break
                }
