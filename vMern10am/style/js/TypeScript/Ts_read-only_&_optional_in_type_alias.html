TS Readonly and optional properties in Obj-Type alias

#Ts Readonly and optional properties in Obj-Type alias :

  1. Use of 'readonly' and ? in type alias .
  2. It is either for Interface.

            
Example :

Obj-Type Alias :
                type EmployeeType = {
                        readonly id: number,
                        name: string,
                        dsg: string,
                        salary: number,
                        city?: string,
                        state?: string,
                }
                var ob1:EmployeeType =  {
                        id:1001,
                        name:"Nitin Chauhan",
                        dsg:"Trainer",
                        salary:135000
                }
                var ob2:EmployeeType =  {
                        id:1002,
                        name:"Deepak Singh",
                        dsg:"Trainer",
                        salary:85000,
                        city:"Ghaziabad",
                        state:"UP"
                }

                console.log(obj1)
                console.log(obj2)



ReadOnly and optional property in interface

                interface EmployeeType{
                        readonly id: number,
                        name: string,
                        dsg: string,
                        salary: number,
                        city?: string,
                        state?: string,
                }
                var ob1:EmployeeType =  {
                        id:1001,
                        name:"Nitin Chauhan",
                        dsg:"Trainer",
                        salary:135000
                }
                var ob2:EmployeeType =  {
                        id:1002,
                        name:"Deepak Singh",
                        dsg:"Trainer",
                        salary:85000,
                        city:"Ghaziabad",
                        state:"UP"
                }
                
                console.log(obj1)
                console.log(obj2)
