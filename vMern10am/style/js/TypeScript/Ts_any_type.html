TS Any dataType

#Ts Any dataType :

=> Means a variable can store different datatype by any-datatype. 
            

            Example :

                        var a: any = 10
                        console.log(a)
                        a = "Ducat"
                        console.log(a)
                        a = true
                        console.log(a)
                        a = [10, 20, 30, 40, 50, 60]
                        console.log(a)

=============================[Simple interest]======================================================
                       
                        var p:number = 135000
                        var t:number = 5
                        var r:number = 12.36
                        var si:number = p*r*t/100
                        var total:number = p+si
                        var emi = total/(t*12)
                        console.log(`
                                Principal Amount = ${p}
                                Rate             = ${r}
                                Time             = ${t}
                                Simple Interest  = ${si.toFixed(2)}
                                Total Amount     = ${total.toFixed(2)}
                                EMI Amount       = ${emi.toFixed(2)}
                            `)
